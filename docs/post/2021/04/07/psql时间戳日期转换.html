postgres时间戳日期转换

2021年4月7日
| postgres
| 阅读 
# 获取当前时间戳
SELECT EXTRACT(epoch FROM NOW());
# 日期转时间戳
SELECT EXTRACT(epoch FROM CAST('2020-05-06 00:30:10' AS TIMESTAMP));
# 时间戳转日期
SELECT TO_TIMESTAMP(1589178132.43633); 
# 时间戳转date
select to_date(to_timestamp(1588725010)::text,'YYYY-MM-DD');
